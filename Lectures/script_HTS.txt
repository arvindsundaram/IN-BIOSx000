# Move to $HOME using "cd" and then move to Desktop
# Create a new folder and "link" data from the shared area.
$ cd
$ cd Desktop
$ mkdir 01-fastqc
$ cd 01-fastqc
$ ln /data/arvind/01_preprocessing/01_fastqc_hands_on/*fq .
# This should link 4 fastq.gz files: raw_1.fastq.gz, raw_2.fastq.gz, raw_bad_1.fastq.gz, raw_bad_2.fastq.gz

# To run FastQC on the above four files:
$ fastqc raw_1.fastq.gz raw_2.fastq.gz raw_bad_1.fastq.gz raw_bad_2.fastq.gz

$ cd
$ cd Desktop
$ mkdir 02-trim
$ cd 02-trim
$ ln /data/arvind/01_preprocessing/02_trimmomatic_hands_on/* .
# This should link 3 files: 1M_R1.fastq.gz, 1M_R2.fastq.gz, script.sh

# You will find the command to run trimmomatic in the script.sh file
$ cat script.sh

# To run trimmomatic on the above files:
$ java -jar /share/inf-biox121/trimmomatic/Trimmomatic-0.36/trimmomatic-0.36.jar PE -phred33 -trimlog trim_log 1M_R1.fastq.gz 1M_R2.fastq.gz trim_R1.fq.gz trim_R1_un.fq.gz trim_R2.fq.gz trim_R2_un.fq.gz ILLUMINACLIP:/share/inf-biox121/trimmomatic/Trimmomatic-0.36/adapters/TruSeq3-PE-2.fa:2:30:10 LEADING:3 TRAILING:3 SLIDINGWINDOW:4:15 MINLEN:36 CROP:125 2> trim_log2 &

# check trim_log2 for output information
$ cat trim_log2

# Run FastQC on original file and trimmomatic output to see what has happened:
$ fastqc 1M_R1.fastq.gz 1M_R2.fastq.gz trim_R1.fq.gz trim_R1_un.fq.gz trim_R2.fq.gz trim_R2_un.fq.gz
